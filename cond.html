channel apply semaphore-使用Channel实现信号量

func GetSqrt(n int, name string, chSem chan string) {
	chSem <- name
	sqrt := math.Sqrt(float64(n))
	fmt.Printf("%d 平方=%f.2f\n", n, sqrt)
	<-chSem
}
func main() {
	chSem := make(chan string, 5)
	for i := 0; i < 50; i++ {
		go GetSqrt(i, strconv.Itoa(i), chSem)
	}
	time.Sleep(4 * time.Second)
}
