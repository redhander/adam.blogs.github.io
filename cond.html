Using sync.Cond 使用sync.Cond


var (
	cond   = sync.Cond{L: &sync.Mutex{}}
	maxNum = 15
)

func main() {

	for i := 0; i < 10; i++ {
		go func(i int) {
			cond.L.Lock()
			defer cond.L.Unlock()
			cond.Wait() //unless cond is notified or broadcast, it will stuck here.
			fmt.Printf("goroutine:%d\n", i)
			time.Sleep(time.Millisecond * 100)
		}(i)
	}
	time.Sleep(3 * time.Second)
	cond.L.Lock()
	maxNum = 16
	println("maxNum is 16 now")
	cond.L.Unlock()
	cond.Broadcast() //cond.Signal. signal randomly select one waiting goroutine to notify. Broadcast notifies all
	time.Sleep(3 * time.Second)
}
