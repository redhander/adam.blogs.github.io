func main() {
	resultCh := make(chan int, 10)
	for i := 0; i < cap(resultCh); i++ {
		go Task(resultCh, i)
	}
	for j := 0; j < cap(resultCh); j++ {
		fmt.Printf("get resultCh:%d\n", <-resultCh)
	}
}
func Task(resultCh chan<- int, param int) {
	//println("start task")
	time.Sleep(time.Second)
	//println("finish task")
	resultCh <- param
}
